   =goal>

      done        nil
==>

	isa   copying-span        
   =goal>
      state       start

(P find-unattended-word
   =goal>
      ISA         goal-mobile-typing
      state       start
      done        nil
   =imaginal>
	isa   copying-span
 ==>
   =imaginal>   ;; prevent strict harvesting
   +visual-location>
      ISA         visual-location
      :attended    nil
	screen-y	lowest
	screen-x	lowest

   =goal>
      state       find-location

(P attend-word
   =goal>
      ISA         goal-mobile-typing
      state       find-location
   =visual-location>
      ISA         visual-location
   
   ?visual>
      state        free
   
==>
   +visual>
      ISA         move-attention
      screen-pos  =visual-location
   =goal>
      state       attend
)



(P encode-word1
   =goal>
      ISA         goal-mobile-typing
      state       attend
   =visual>
      ISA         text
      value       =letters		;; with quotation marks e.g., "test "

   =visual>
      ISA         text
      - value       "." 		;; not the end

   =imaginal>
	isa   copying-span
        word1    nil
==>
!bind! =letters (replace-space-in-string  =letters  _ )          ;; "_" is used in the system to represent space
   =goal>
      state       start
      remaining-letters      =letters  

   =imaginal>
        word1    =letters
   +visual> 
      isa clear   ;;disable re-encoding, -visual> or strict harvesting can clear visual buffer, but cannot clear last attended info to disable re-encoding.
)


(P encode-word2
   =goal>
      ISA         goal-mobile-typing
      state       attend
      remaining-letters      =previous-letters  	
   =visual>
      ISA         text
      value       =letters		;; with quotation marks e.g., "test "

   =visual>
      ISA         text
      - value       "." 		;; not the end

   =imaginal>
	isa   copying-span
        - word1    nil
	word2   nil
==>
!bind! =letters (replace-space-in-string  =letters  _ )    ;; "_" is used in the system to represent space
!bind! =letters (combine-two-strings-with-quotation-marks  =previous-letters  =letters)    ;; this is the current way to read individual words from the sentence visual text object.

   =goal>
      state       respond
      remaining-letters      =letters
   =imaginal>
        word2    =letters
   +visual> 
      isa clear   ;;disable re-encoding, -visual> or strict harvesting can clear visual buffer, but cannot clear last attended info to disable re-encoding.
)

==>

   =goal>

   +visual> 
      isa clear   ;;disable re-encoding, -visual> or strict harvesting can clear visual buffer, but cannot clear last attended info to disable re-encoding.
)



(p typing-letter-start
   =goal>
      ISA         goal-mobile-typing
      state       respond
      remaining-letters	  =letters    ;; not nil	

   =goal>
      ISA         goal-mobile-typing
      - remaining-letters	  "_"	

==>
!bind! =letters (remove-quotation-marks  =letters )        ;; this is the current way to obtain individual letters from a word
!bind! =first ( add-quotation-marks (get-char-from-string  =letters  1) )
!bind! =rest  ( add-quotation-marks (get-tail-from-string  =letters  2) )
   =goal>
	state 		typing
	typing-letter	=first
	remaining-letters =rest	
)
